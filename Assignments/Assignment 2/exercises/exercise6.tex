\begin{align}
	&	Cl \rightarrow Cl; n : St \\
	& \dsstmt{Cl\text{ }e} = \dscond{\dsbool{n==e}}
																 	{\dsstmt{St}}
								 									{\dsstmt{Cl\text{ }e}}
\end{align}
And therefor:
\begin{align}
	& \dsstmt{Cl\text{ }e} \\
= & \dsfix{F} \\
	& \text{where } F g = \dscond{\dsbool{n==e}}
															 {\dsstmt{St}}
															 {g} 
\end{align}

	& Cl \rightarrow n : St \\
	& \dsstmt{Cl\text{ }e} = \dscond{\dsbool{n==e}}
								 						{\dsstmt{St}}
								 						{id}
\end{align}
